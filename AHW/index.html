Parpadeo de un LED

						
Primero se configura el pin deseado como salida

						
							void setup() {
								pinMode(9,OUTPUT); // Configura el pin 9 como salida
							}
							void loop() {
								digitalWrite(9, HIGH); // Activa el pin 9 con 5V
								delay(500);			   // Detiene la ejecución 500 ms
								digitalWrite(9, LOW);  // Desactiva el pin 0  0V
								delay(500);			   // Detiene la ejecución 500 ms
							}
